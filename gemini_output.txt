                font-family: 'Fira Code', monospace;

            }

            @keyframes fadeIn {

    passwordInput.type = "password";
    passwordInput.placeholder = "Enter access key...";
    passwordInput.autofocus = true;
    modalBox.appendChild(passwordInput);

    const errorDisplay = document.createElement("div");
    errorDisplay.className = "modal-error";
    errorDisplay.style.visibility = "hidden"; // Use visibility to reserve space
    modalBox.appendChild(errorDisplay);

    const submitButton = document.createElement("button");
    submitButton.type = "submit";
    submitButton.textContent = "[ ACCESS ]";
    modalBox.appendChild(submitButton);

    const cancelText = document.createElement("p");
    cancelText.className = "cancel-text";
    const cancelSpan = document.createElement("span");
    cancelSpan.textContent = "[ CANCEL ]";
    cancelSpan.addEventListener("click", onClose);
    cancelText.appendChild(cancelSpan);
    modalBox.appendChild(cancelText);

    overlay.appendChild(modalBox);

    // Event listener for form submission (or button click)
    submitButton.addEventListener("click", async (e) => {
        e.preventDefault(); // Prevent default form submission if it were a form
        errorDisplay.style.visibility = "hidden";
        errorDisplay.textContent = "";

        const password = passwordInput.value;
        if (!password) {
            errorDisplay.textContent = "// PASSWORD REQUIRED";
            errorDisplay.style.visibility = "visible";
            return;
        }

        try {
            const res = await fetch("/api/auth/login", {
                method: "POST",
                headers: { "Content-Type": "application/json" },
                body: JSON.stringify({ password })
            });

            if (res.status === 200) {
                const data = await res.json();
                onLogin(data.token);
                onClose(); // Close modal on successful login
            } else if (res.status === 401) {
                errorDisplay.textContent = "// INVALID CREDENTIALS";
                errorDisplay.style.visibility = "visible";
            } else {
                errorDisplay.textContent = "// AUTHENTICATION FAILED. TRY AGAIN.";
                errorDisplay.style.visibility = "visible";
            }
        } catch (error) {
            console.error("Login fetch error:", error);
            errorDisplay.textContent = "// CONNECTION FAILED. SERVER OFFLINE?";
            errorDisplay.style.visibility = "visible";
        }
    });

    // Close modal if clicking outside the modal box
    overlay.addEventListener("click", (e) => {
        if (e.target === overlay) {
            onClose();
        }
    });

    // Close modal with Escape key
    const handleKeydown = (e) => {
        if (e.key === "Escape") {
            onClose();
            document.removeEventListener("keydown", handleKeydown); // Clean up
        }
    };
    document.addEventListener("keydown", handleKeydown);
    // Ensure cleanup is done on close
    const originalOnClose = onClose;
    onClose = () => {
        document.removeEventListener("keydown", handleKeydown);
        originalOnClose();
    };


    return overlay;
}

    ]);

    if (!swRes.ok) throw new Error(`HTTP error! status: ${swRes.status}`);
    if (!catRes.ok) throw new Error(`HTTP error! status: ${catRes.status}`);

    allSoftware = await swRes.json();
    categories = await catRes.json();
  } catch (error) {
    console.error("Failed to load initial data:", error);
    alert("// ERROR: Could not load data. Backend offline?");
    allSoftware = []; // Clear data to avoid rendering stale/corrupted info
    categories = [];
  }
}

function render() {
  const root = document.getElementById("app");
  root.innerHTML = ""; // Clear existing content

  if (isLoggedIn) {
    // Re-render admin panel if logged in
    adminPanelInstance = renderAdminPanel(onAdd, logout);
    root.appendChild(adminPanelInstance.element);
  } else {
    adminPanelInstance = null; // Clear instance if logged out
  }

  root.appendChild(renderCategoryFilter(categories, activeCategory, onCategorySelect));

  const filtered = activeCategory === "All" ? allSoftware : allSoftware.filter(s => s.category === activeCategory);
  root.appendChild(renderSoftwareGrid(filtered, onDelete, isLoggedIn));
}

function onCategorySelect(cat) {
  activeCategory = cat;
  render();
}

function showLoginModal() {
  const modal = renderLoginModal(onLogin, () => {
    modal.remove(); // Remove modal from DOM when closed
  });
  document.body.appendChild(modal);
}

function onLogin(token) {
  sessionStorage.setItem("sw_token", token);
  authToken = token;
  isLoggedIn = true;
  document.getElementById("login-btn").textContent = "[ LOGOUT ]";
  render(); // Re-render to show admin panel
}

function logout() {
  sessionStorage.removeItem("sw_token");
  authToken = null;
  isLoggedIn = false;
  document.getElementById("login-btn").textContent = "[ LOGIN ]";
  render(); // Re-render to hide admin panel
}

async function onAdd(data) {
  try {
    const res = await fetch("/api/software", {
      method: "POST",
      headers: { 
        "Content-Type": "application/json", 
        "Authorization": "Bearer " + authToken 
      },
      body: JSON.stringify(data)
    });

    if (res.status === 201) {
      await loadData(); // Reload all software to include the new one
      render();
      if (adminPanelInstance) adminPanelInstance.resetForm(); // Clear form
    } else if (res.status === 401) {
      alert("// SESSION EXPIRED. PLEASE LOG IN AGAIN.");
      logout(); // Log out if unauthorized
    } else {
      const err = await res.json();
      alert("// ERROR: " + (err.error || "Failed to add software"));
    }
  } catch (error) {
    console.error("Error adding software:", error);
    alert("// ERROR: Connection failed or server error.");
  }
}

async function onDelete(id) {
  if (!confirm("// CONFIRM DELETION OF SOFTWARE? THIS ACTION IS IRREVERSIBLE.")) return; // Confirmation

  try {
    const res = await fetch("/api/software/" + id, {
      method: "DELETE",
      headers: { 
        "Authorization": "Bearer " + authToken 
      }
    });

    if (res.status === 200) {
      await loadData(); // Reload software list
      render();
    } else if (res.status === 401) {
      alert("// SESSION EXPIRED. PLEASE LOG IN AGAIN.");
      logout(); // Log out if unauthorized
    } else {
      const err = await res.json();
      alert("// ERROR: " + (err.error || "Failed to delete software"));
    }
  } catch (error) {
    console.error("Error deleting software:", error);
    alert("// ERROR: Connection failed or server error.");
  }
}

document.addEventListener("DOMContentLoaded", async () => {
  const loginBtn = document.getElementById("login-btn");
  loginBtn.textContent = isLoggedIn ? "[ LOGOUT ]" : "[ LOGIN ]";
  
  loginBtn.addEventListener("click", () => {
    if (isLoggedIn) {
      logout();
    } else {
      showLoginModal();
    }
  });

  await loadData();
  render();
});
